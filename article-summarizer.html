# Load spaCy and make sure sentiment analyzer is added
nlp = spacy.load("en_core_web_sm")
if "spacytextblob" not in nlp.pipe_names:
    nlp.add_pipe("spacytextblob", last=True)

# Read the HTML file and pull the article text
with open("stardew_review.html", "r", encoding="utf-8") as file:
    html = file.read()

soup = BeautifulSoup(html, "html.parser")
article = soup.find("div", class_="c-entry-content")
text = article.get_text(separator="\n", strip=True) if article else soup.get_text(separator="\n", strip=True)

# Process the full article
doc = nlp(text)

# Go through each sentence, lemmatize it, and get the sentiment score
lemma_scores = []
for sent in doc.sents:
    # Turn the sentence into a string of its lemmas (stripped of punctuation/spacing)
    lemmatized = " ".join([token.lemma_ for token in sent if not token.is_space])
    # Run it through the pipeline again to get the sentiment
    score = nlp(lemmatized)._.blob.polarity
    lemma_scores.append(score)

# Plot the histogram
plt.figure(figsize=(10, 6))
plt.hist(lemma_scores, bins=20, edgecolor="black")
plt.title("Sentiment Scores Based on Lemmatized Sentences")
plt.xlabel("Polarity Score")
plt.ylabel("Number of Sentences")
plt.grid(axis='y', alpha=0.75)
plt.tight_layout()
plt.show()

# Most of the sentences seem to fall between 0.0 and 0.2 again,
# so overall it's still coming across as neutral to mildly positive.
